## ðŸ” Smart Contract Design

### ContextStorageRegistry (Updated)

```solidity
contract ContextStorageRegistry {

    // Storage backing ratio: 1 PSI = 1 MB
    uint256 public constant PSI_PER_MB = 1e18; // 1 PSI (18 decimals)
    uint256 public constant BYTES_PER_PSI = 1_000_000; // 1 MB

    struct ContextNode {
        bytes32 ipfsCID;
        uint256 sizeBytes;
        uint256 lockedPSI;      // Tokens locked as collateral
        uint256 rentDeposit;    // Separate rent payment pool
        uint256 lastRentCharge;
        address owner;
        bool archived;
    }

    mapping(bytes32 => ContextNode) public contexts;

    // Track total locked vs liquid supply
    uint256 public totalLockedPSI;

    /**
     * @notice Register context and lock PSI tokens
     */
    function registerContext(
        bytes32 ipfsCID,
        uint256 sizeBytes,
        uint256 rentDepositAmount
    ) external {
        // Calculate required locked tokens
        uint256 requiredPSI = (sizeBytes * PSI_PER_MB) / BYTES_PER_PSI;

        // Lock storage tokens (collateral)
        psiToken.transferFrom(msg.sender, address(this), requiredPSI);

        // Separate rent deposit
        psiToken.transferFrom(msg.sender, address(this), rentDepositAmount);

        contexts[ipfsCID] = ContextNode({
            ipfsCID: ipfsCID,
            sizeBytes: sizeBytes,
            lockedPSI: requiredPSI,
            rentDeposit: rentDepositAmount,
            lastRentCharge: block.timestamp,
            owner: msg.sender,
            archived: false
        });

        totalLockedPSI += requiredPSI;

        emit ContextRegistered(ipfsCID, msg.sender, sizeBytes, requiredPSI);
    }

    /**
     * @notice Monthly rent: 0.1% of locked tokens
     */
    function calculateMonthlyRent(bytes32 cid) public view returns (uint256) {
        ContextNode storage context = contexts[cid];
        // 0.1% per month = 10 basis points
        return (context.lockedPSI * 10) / 10000;
    }

    /**
     * @notice Optimize context and unlock tokens
     */
    function acceptOptimization(
        bytes32 originalCID,
        uint256 newSize,
        uint256 qualityScore
    ) internal {
        ContextNode storage context = contexts[originalCID];

        uint256 oldLockedPSI = context.lockedPSI;
        uint256 newLockedPSI = (newSize * PSI_PER_MB) / BYTES_PER_PSI;
        uint256 freedPSI = oldLockedPSI - newLockedPSI;

        // Update locked amount
        context.lockedPSI = newLockedPSI;
        context.sizeBytes = newSize;
        totalLockedPSI -= freedPSI;

        // Calculate efficiency score
        uint256 compressionRatio = (context.sizeBytes * 10000) / newSize;
        uint256 efficiencyScore = (qualityScore * compressionRatio) / 10000;

        // Unlock freed tokens back to agent
        psiToken.transfer(context.owner, freedPSI);

        // Efficiency bonus from pool
        if (efficiencyScore > 100) {
            uint256 bonus = ((efficiencyScore - 100) * freedPSI) / 1000;
            if (bonus > 0 && bonus <= efficiencyBonusPool) {
                efficiencyBonusPool -= bonus;
                psiToken.transfer(context.owner, bonus);
            }
        }

        emit TokensUnlocked(originalCID, freedPSI, efficiencyScore);
    }

    /**
     * @notice Archive context and unlock all tokens
     */
    function archiveContext(bytes32 cid) external {
        ContextNode storage context = contexts[cid];
        require(context.owner == msg.sender, "Not owner");

        uint256 refundPSI = context.lockedPSI;
        uint256 refundDeposit = context.rentDeposit;

        // Mark as archived
        context.archived = true;
        context.lockedPSI = 0;
        context.rentDeposit = 0;
        totalLockedPSI -= refundPSI;

        // Refund all locked tokens + remaining rent
        psiToken.transfer(msg.sender, refundPSI + refundDeposit);

        emit ContextArchived(cid, refundPSI);
    }

    /**
     * @notice Get current network utilization
     * @return Percentage of total supply locked (0-10000 basis points)
     */
    function getNetworkUtilization() external view returns (uint256) {
        uint256 totalSupply = psiToken.totalSupply();
        return (totalLockedPSI * 10000) / totalSupply;
    }
}
```
